конфигурация = объект с настройками игры:
    гравитация = 0.8
    скорость_прыжка = -14
    скорость_движения = 1.2
    макс_скорость = 9
    трение = 0.82
    размер_тайла = 32
    столбцов = 50
    строк = 30
    всего_монет = 20
    число_врагов = 10
    цвета_игрока = объект с цветами тела и свечения
    цвета_врага = объект с цветами тела и свечения
    цвета_монеты = массив из четырёх оттенков
    оттенок_блока = 210

класс МИР это
    свойства: конфиг, размер_тайла, столбцы, строки, сетка, монеты, всего_монет, зерно
    метод СОЗДАТЬ() это
        заполнить сетку нулями
        установить нижние две строки как блоки (1)
        в цикле 90 раз:
            выбрать случайные координаты и размеры платформы (2-4 тайла в ширину, 1-2 в высоту)
            если платформа помещается внутри поля и не выходит за нижние две строки
                то заполнить соответствующую область сетки единицами
        в цикле всего_монет раз:
            выбрать случайную пустую клетку (не блок)
            установить в ней значение 3 (монета)
        вернуть сетку
    конец СОЗДАТЬ

    метод ГСЧ() это
        обновить зерно по формуле линейного конгруэнтного генератора
        вернуть дробное число от 0 до 1
    конец ГСЧ

    метод ПОЛУЧИТЬ_ТАЙЛ(колонка, строка) это
        если координаты вне поля то вернуть 1 (блок)
        иначе вернуть значение из сетки
    конец ПОЛУЧИТЬ_ТАЙЛ

    метод ЭТО_БЛОК(колонка, строка) это вернуть (ПОЛУЧИТЬ_ТАЙЛ == 1)
    метод ЭТО_МОНЕТА(колонка, строка) это вернуть (ПОЛУЧИТЬ_ТАЙЛ == 3)

    метод СОБРАТЬ_МОНЕТУ(колонка, строка) это
        если ЭТО_МОНЕТА истина то
            обнулить клетку в сетке
            уменьшить счётчик монет
            вернуть истина
        иначе вернуть ложь
    конец СОБРАТЬ_МОНЕТУ
конец МИР

класс СУЩНОСТЬ это
    свойства: х, у, ширина, высота, скорость_х, скорость_у, конфиг, спавн
    метод ОБНОВИТЬ_ФИЗИКУ(мир) это
        скорость_у += гравитация
        х += скорость_х
        если СТОЛКНОВЕНИЕ(мир) то откатить х и обнулить скорость_х
        у += скорость_у
        если СТОЛКНОВЕНИЕ(мир) то
            если скорость_у > 0 то
                установить у на границу блока снизу (округление вниз) и обнулить скорость_у
            иначе
                установить у на границу блока сверху и обнулить скорость_у
        ограничить координаты х и у границами поля (0 .. ширина_поля - ширина, 0 .. высота_поля - высота)
    конец ОБНОВИТЬ_ФИЗИКУ

    метод СТОЛКНОВЕНИЕ(мир) это
        вычислить диапазон тайлов, которые перекрывает сущность
        перебрать все тайлы в диапазоне
        если какой-либо тайл является блоком (1) то вернуть истина
        вернуть ложь
    конец СТОЛКНОВЕНИЕ

    метод НА_ПОЛУ(мир) это
        проверить, есть ли блок непосредственно под ногами (на одну строку ниже) с учётом допуска 6 пикселей
    конец НА_ПОЛУ

    метод НА_ПОТОЛКЕ(мир) это
        проверить, есть ли блок непосредственно над головой с допуском 6 пикселей
    конец НА_ПОТОЛКЕ
конец СУЩНОСТЬ

класс ИГРОК расширяет СУЩНОСТЬ это
    свойства: мёртв, таймер_воскрешения
    метод ИНИЦИАЛИЗИРОВАТЬ(мир) это
        попытаться найти свободную позицию (не пересекающую блоки) в поле, перебирая случайные координаты до 30 попыток
        если найдена, установить координаты, обнулить скорости, сбросить мёртв и спавн
        иначе установить позицию по умолчанию (32, строка-2)*размер_тайла - 32
    конец ИНИЦИАЛИЗИРОВАТЬ

    метод ОБНОВИТЬ(ввод, мир) это
        если мёртв то выйти
        если спавн > 0 (идёт анимация появления) то увеличить спавн и выйти
        рассчитать ускорение по горизонтали на основе нажатых клавиш (влево/вправо)
        применить трение и ограничить максимальную скорость
        вызвать ОБНОВИТЬ_ФИЗИКУ(мир)
        если нажата клавиша прыжка и персонаж на полу или потолке, то задать вертикальную скорость (прыжок с учётом боковой скорости)
        если у + высота > высота_поля, то установить мёртв = истина и обнулить таймер_воскрешения
    конец ОБНОВИТЬ

    метод ВОСКРЕСИТЬ(мир) это
        увеличить таймер_воскрешения
        если таймер > 55 то вызвать ИНИЦИАЛИЗИРОВАТЬ(мир) и вернуть истина
        вернуть ложь
    конец ВОСКРЕСИТЬ
конец ИГРОК

класс ВРАГ расширяет СУЩНОСТЬ это
    свойства: таймер, ожидание, цель_х, цель_у, миграция, спавн
    метод НАЙТИ_СПАВН(мир) это
        до 50 попыток найти свободную позицию (не блок)
        установить случайную начальную скорость по х
        обнулить скорость_у
        установить цель_х и цель_у равными текущим координатам
        обнулить миграцию
        спавн = 1
    конец НАЙТИ_СПАВН

    метод ОБНОВИТЬ(мир) это
        если спавн > 0 то анимировать появление и выйти
        увеличить таймер
        если таймер > ожидание то
            сбросить таймер, новое ожидание (5-20)
            выбрать случайное направление движения (смещение) и установить цель_х, цель_у на расстоянии 40-250 по х и 30-200 по у
            задать миграцию = 40-90
            с вероятностью 80% задать вертикальную скорость для прыжка (-8..-4)
        если миграция > 0 то
            уменьшить миграцию
            вычислить направление к цели, применить ускорение к скорости_х с затуханием
            если приблизились к цели по х, то сбросить миграцию
            если приблизились по у, то прыгнуть
        иначе (миграция == 0) то
            выбрать новую случайную цель с меньшим радиусом и задать миграцию 30-60
        ограничить скорость_х по модулю макс_скоростью
        вызвать ОБНОВИТЬ_ФИЗИКУ(мир)
    конец ОБНОВИТЬ
конец ВРАГ

класс КАМЕРА это
    свойства: х, у, ширина, высота, конфиг
    метод СЛЕДОВАТЬ(сущность) это
        установить х = сущность.х - ширина/2
        установить у = сущность.у - высота/2
        ограничить х и у границами мира (0 .. ширина_мира - ширина_камеры, 0 .. высота_мира - высота_камеры)
    конец СЛЕДОВАТЬ
конец КАМЕРА

класс СИСТЕМА_ЧАСТИЦ это
    свойства: частицы, следы
    метод ВЗРЫВ(х, у, цвет, количество=35, монета=ложь) это
        создать количество частиц со случайными направлениями, скоростями, временем жизни и размером
        если монета истина то увеличить размер и время жизни
    конец ВЗРЫВ

    метод ФОНТАН_МОНЕТ(х, у) это
        создать 60 частиц в виде монет с разбросом скоростей и большим временем жизни
    конец ФОНТАН_МОНЕТ

    метод ДОБАВИТЬ_СЛЕД(х, у, цвет) это
        добавить частицу-след с временем жизни 40
    конец ДОБАВИТЬ_СЛЕД

    метод ОБНОВИТЬ() это
        для каждой частицы: обновить позицию, применить гравитацию, уменьшить жизнь, удалить мёртвые
        для каждого следа: уменьшить жизнь, удалить мёртвые
    конец ОБНОВИТЬ

    метод ОЧИСТИТЬ() это очистить оба массива
конец СИСТЕМА_ЧАСТИЦ

класс УПРАВЛЕНИЕ_ЗВЁЗДАМИ это
    свойства: звёзды, время
    метод ДОБАВИТЬ_ЗВЕЗДУ() это добавить новый объект с углом и фазой
    метод ОБНОВИТЬ(дельта) это увеличить время, обновить углы и фазы всех звёзд
    метод ОЧИСТИТЬ() это очистить массив
конец УПРАВЛЕНИЕ_ЗВЁЗДАМИ

класс ВВОД это
    свойства: клавиши, влево, вправо, прыжок, чит

        подписаться на события клавиатуры (keydown/keyup)

    метод ОБНОВИТЬ_СОСТОЯНИЕ() это
        установить влево = нажат ArrowLeft или A

    метод СБРОСИТЬ_ЧИТ() это чит = ложь
конец ВВОД

класс ОТРИСОВЩИК это
    свойства: холст, контекст, конфиг, ширина, высота
    метод ИЗМЕНИТЬ_РАЗМЕР() это
        подогнать размер холста под окно с сохранением пропорций 800x500
    метод ОЧИСТИТЬ() это
        залить фон градиентом с плавно меняющимися цветами в зависимости от времени
    метод НАРИСОВАТЬ_МИР(мир, камера) это
        вычислить видимую область (по тайлам)
        для каждого видимого тайла:
            если блок (1) то нарисовать его с градиентом и тенью, оттенок меняется от времени и координат
            если монета (3) то нарисовать блестящую монету с кругами и бликами
    метод НАРИСОВАТЬ_ЧАСТИЦЫ(система, камера) это
        для каждой частицы: нарисовать окружность с учётом прозрачности (жизни) и свечения, для монет увеличенный размер
        для каждого следа: нарисовать полупрозрачную окружность
    метод НАРИСОВАТЬ_СЛАЙМ(х, у, ширина, высота, состояние, игрок, спавн, конфиг, цель_монета=нуль) это
        с учётом состояния (на полу, на потолке, в воздухе) и анимации спавна нарисовать эллипс с градиентом
        добавить тень и обводку
        если игрок и есть ближайшая монета, нарисовать глаза, смотрящие на монету, иначе стандартные глаза (белки и зрачки)
        для врагов аналогично с красными цветами
    метод НАРИСОВАТЬ_ЗВЁЗДЫ(звёзды, игрок, камера, время) это
        для каждой звезды вычислить позицию вокруг игрока с колебаниями и нарисовать пятиконечную звезду
    метод НАРИСОВАТЬ_ЗВЕЗДУ(центр_х, центр_у, радиус, цвет, поворот=0, лучей=5) это
        нарисовать пятиконечную звезду с заданными параметрами
    метод НАРИСОВАТЬ_ИНТЕРФЕЙС(собрано, всего, победа, таймер_победы, эмодзи) это
        вывести счёт монет в левом верхнем углу
        если победа и таймер между 20 и 120, то отобразить анимированный эмодзи 🎉 с увеличением и прозрачностью
конец ОТРИСОВЩИК

класс ИГРА это
    свойства: конфиг, холст, отрисовщик, мир, игрок, камера, частицы, звёзды, ввод, враги, собрано_монет, всего_монет, победа, таймер_победы, эмодзи_победы, игрок_мёртв, взрыв, таймер_воскрешения, время, запущена, события
    конструктор(холст, конфиг):
        объединить переданный конфиг с базовым
        инициализировать все компоненты
        подписаться на изменение размера окна
        вызвать ГЕНЕРИРОВАТЬ_УРОВЕНЬ
    метод ГЕНЕРИРОВАТЬ_УРОВЕНЬ() это
        создать мир (вызвать СОЗДАТЬ)
        сбросить счётчики и флаги
        инициализировать игрока
        создать врагов (число_врагов) через Enemy
        добавить начальные фоновые частицы (40 штук)
    метод ЗАПУСТИТЬ() это
        если не запущена, установить флаг и вызвать ЦИКЛ
    метод ОСТАНОВИТЬ() это установить запущена = ложь
    метод ЦИКЛ() это
        если не запущена то выйти
        вызвать ОБНОВИТЬ и ОТРИСОВАТЬ
        запросить следующий кадр через requestAnimationFrame
    метод ОБНОВИТЬ() это
        обновить время
        обработать чит (если активирован, собрать все монеты на карте)
        если победа, то увеличить таймер, запустить фейерверк при первом кадре, добавить новую звезду и перегенерировать уровень через 120 кадров
        если игрок мёртв, то создать взрыв частиц, увеличить таймер воскрешения, при достижении 55 перегенерировать уровень
        иначе обновить игрока; если он умер, вызвать событие смерти и выйти
        проверить сбор монеты на позиции игрока (если есть, собрать, увеличить счёт, вызвать событие, при достижении всех монет установить победу)
        обновить всех врагов; проверить столкновения с игроком (если есть, убить игрока)
        обработать столкновения врагов друг с другом: если пересекаются, взорвать обоих, удалить, создать двух новых врагов взамен
        поддерживать количество врагов не менее заданного (добавлять новых)
        добавить следы за игроком при движении
        обновить камеру (следовать за игроком)
        обновить частицы и звёзды
    метод ОТРИСОВАТЬ() это
        очистить холст
        нарисовать мир с учётом камеры
        нарисовать частицы
        нарисовать всех врагов (с определением состояния на полу/потолке)
        нарисовать игрока (если не мёртв и не победа), с определением состояния и поиском ближайшей монеты для направления взгляда
        нарисовать звёзды и интерфейс
    метод ПОДПИСАТЬСЯ(событие, колбэк) это добавить колбэк в массив события
    метод ВЫЗВАТЬ_СОБЫТИЕ(событие, ...аргументы) это выполнить все колбэки для данного события
конец ИГРА
